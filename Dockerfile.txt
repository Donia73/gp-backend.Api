# Use the official .NET 8 SDK image for build stage
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy the solution file and projects
COPY ["YourSolution.sln", "./"]
COPY ["AuthenticationServer.Api/AuthenticationServer.Api.csproj", "AuthenticationServer.Api/"]
COPY ["gp-backend.Api/gp-backend.Api.csproj", "gp-backend.Api/"]
COPY ["gp-backend.EF/gp-backend.EF.csproj", "gp-backend.EF/"]
COPY ["gp-backend.Core/gp-backend.Core.csproj", "gp-backend.Core/"]

# Restore dependencies
RUN dotnet restore "YourSolution.sln"

# Copy the rest of the files and build the projects
COPY . .
WORKDIR "/src/AuthenticationServer.Api"
RUN dotnet publish -c Release -o /app/publish

WORKDIR "/src/gp-backend.Api"
RUN dotnet publish -c Release -o /app/publish

# Use the official .NET 8 ASP.NET Core Runtime image for runtime stage
FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS runtime
WORKDIR /app
COPY --from=build /app/publish .

# Expose the necessary ports
EXPOSE 80
EXPOSE 443

# Set environment variables for the JSON configuration
ENV ASPNETCORE_URLS=http://+:80
ENV ASPNETCORE_ENVIRONMENT=Production

# Run the application
ENTRYPOINT ["dotnet", "AuthenticationServer.Api.dll"]
